    /* ------------- LOADER ------------- */
    const loader = document.getElementById('loader');
    window.addEventListener('load', () => {
      setTimeout(() => {
        loader.classList.add('loader-hide');
        setTimeout(()=> loader.style.display='none', 900);
        // show welcome card after loader fades
        setTimeout(()=> document.getElementById('welcomeCard').classList.add('show'), 160);
      }, 650);
    });

    /* ------------- NAVBAR scrolled effect ------------- */
    const headerInner = document.getElementById('headerInner');
    const header = document.getElementById('siteHeader');
    window.addEventListener('scroll', () => {
      // show glassy inner pill when user scrolls down slightly
      if (window.scrollY > 40) headerInner.classList.add('scrolled'); else headerInner.classList.remove('scrolled');
    });

    /* ------------- MOBILE MENU ------------- */
    function toggleMenu(){
      const mobile = document.getElementById('mobileMenu');
      if (!mobile) return;
      if (mobile.classList.contains('active')) { mobile.classList.remove('active'); mobile.style.right='-320px'; }
      else { mobile.classList.add('active'); mobile.style.right='0'; }
    }
    window.toggleMenu = toggleMenu;

    /* ------------- PHOTO UNFOLD ON SCROLL ------------- */
    const photoMask = document.getElementById('photoMask');
    const photoPanel = document.getElementById('photoPanel');
    const photoCaption = document.getElementById('photoCaption');
    let opened = false;
    if (photoPanel && 'IntersectionObserver' in window) {
      const obs = new IntersectionObserver((entries) => {
        entries.forEach(e => {
          if (e.isIntersecting && !opened) {
            opened = true;
            photoMask.classList.add('open'); // triggers clip-path -> full image
            photoCaption.classList.add('show');
            // slightly relax image scale for effect
            const img = photoMask.querySelector('img');
            if (img) img.style.transform = 'scale(1)';
          }
        });
      }, { threshold: 0.28 });
      obs.observe(photoPanel);
    } else {
      // fallback for ancient browsers
      photoMask.classList.add('open'); photoCaption.classList.add('show');
    }

    /* ------------- EVENTS: reveal + tilt + spark ------------- */
    const cards = document.querySelectorAll('.card');
    // intersection reveal
    const io = new IntersectionObserver((entries, ob) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const el = entry.target;
          setTimeout(()=> {
            el.classList.add('reveal-in');
            el.style.opacity = '1';
            el.style.transform = 'translateY(0)';
          }, (parseInt(el.dataset.index) || 0) * 70 + 120);
          ob.unobserve(el);
        }
      });
    }, { threshold: 0.12 });
    cards.forEach((c, i) => { c.dataset.index = i; io.observe(c); });

    // interactive tilt + spark
    cards.forEach(card => {
      const spark = card.querySelector('.spark');
      card.addEventListener('mousemove', (ev)=> {
        const r = card.getBoundingClientRect();
        const px = ev.clientX - r.left;
        const py = ev.clientY - r.top;
        const cx = (px / r.width) - 0.5;
        const cy = (py / r.height) - 0.5;
        const rx = (-cy) * 6; const ry = (cx) * 6;
        card.style.transform = `perspective(900px) rotateX(${rx}deg) rotateY(${ry}deg) scale(1.02)`;
        if (spark) {
          spark.style.left = px + 'px';
          spark.style.top = py + 'px';
          spark.style.opacity = 1;
          spark.style.transform = 'translate(-50%,-50%) scale(1)';
        }
      });
      card.addEventListener('mouseleave', ()=> {
        card.style.transform = '';
        if (spark) { spark.style.opacity = 0; spark.style.transform = 'translate(-50%,-50%) scale(.4)'; }
      });
      // keyboard accessible click
      card.addEventListener('click', ()=> {
        const t = card.querySelector('h3')?.innerText || 'Event';
        // placeholder behaviour â€” replace with modal / navigation
        alert(t + "\\n\\nAdd full event page or modal here.");
      });
    });

    /* ------------- SPONSOR logos: images later ------------- */

    /* ------------- AMBIENT PARTICLES (lightweight) ------------- */
    (function() {
      const canvas = document.getElementById('ambientParticles');
      if (!canvas) return;
      const ctx = canvas.getContext('2d');
      let w = canvas.width = innerWidth, h = canvas.height = innerHeight;
      const N = Math.max(18, Math.floor((w*h)/150000));
      const ps = [];
      for (let i=0;i<N;i++) ps.push({x:Math.random()*w,y:Math.random()*h,r:Math.random()*1.4+0.5,vx:(Math.random()-0.5)*0.12,vy:(Math.random()-0.5)*0.06,a:Math.random()*0.18+0.06});
      function draw(){
        ctx.clearRect(0,0,w,h);
        for (const p of ps) {
          ctx.beginPath();
          ctx.fillStyle = `rgba(180,230,200,${p.a})`;
          ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
          ctx.fill();
          p.x+=p.vx; p.y+=p.vy;
          if (p.x<-30) p.x=w+30; if (p.x>w+30) p.x=-30;
          if (p.y<-30) p.y=h+30; if (p.y>h+30) p.y=-30;
        }
        requestAnimationFrame(draw);
      }
      window.addEventListener('resize', ()=>{ w=canvas.width=innerWidth; h=canvas.height=innerHeight; });
      draw();
    })();

    /* ------------- CTA Buttons ------------- */

      document.querySelectorAll('*').forEach(n => n.style.transition='none');
    }
